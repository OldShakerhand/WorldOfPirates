    generateHarbors() {
        // TODO: Replace with tile-based harbor placement from world_map.json
        // TEMPORARY: Generate harbors on land edges with max 50% land coverage in radius
        // This will be replaced when harbors have specific coordinates and fixed names in the map data
        
        const harbors = [];
        const TERRAIN = require('./GameConfig').TERRAIN;
        
        // Find land tiles that could be harbor locations
        // A good harbor location is on land but near water (max 50% land in radius)
        const harborRadius = 30; // GameConfig.HARBOR_INTERACTION_RADIUS
        const tileSize = this.worldMap.tileSize;
        
        // Sample potential harbor locations from land tiles
        const potentialLocations = [];
        
        // Check every 10th tile to find land tiles (optimization)
        for (let tileY = 0; tileY < this.worldMap.height; tileY += 10) {
            for (let tileX = 0; tileX < this.worldMap.width; tileX += 10) {
                if (this.worldMap.getTileByGrid(tileX, tileY) === TERRAIN.LAND) {
                    // Convert tile to world coordinates
                    const worldPos = this.worldMap.tileToWorld(tileX, tileY);
                    
                    // Check if this location has good water access (max 50% land in radius)
                    let landCount = 0;
                    let totalCount = 0;
                    const checkRadius = Math.ceil(harborRadius / tileSize);
                    
                    for (let dy = -checkRadius; dy <= checkRadius; dy++) {
                        for (let dx = -checkRadius; dx <= checkRadius; dx++) {
                            const checkTileX = tileX + dx;
                            const checkTileY = tileY + dy;
                            
                            if (checkTileX >= 0 && checkTileX < this.worldMap.width &&
                                checkTileY >= 0 && checkTileY < this.worldMap.height) {
                                totalCount++;
                                if (this.worldMap.getTileByGrid(checkTileX, checkTileY) === TERRAIN.LAND) {
                                    landCount++;
                                }
                            }
                        }
                    }
                    
                    const landPercentage = landCount / totalCount;
                    
                    // Good harbor location: on land but with water access (20-50% land)
                    if (landPercentage > 0.2 && landPercentage <= 0.5) {
                        potentialLocations.push(worldPos);
                    }
                }
            }
        }
        
        // Select up to 7 harbors from potential locations (spread out)
        const minDistance = 150; // Minimum distance between harbors
        for (let i = 0; i < potentialLocations.length && harbors.length < 7; i++) {
            const pos = potentialLocations[i];
            
            // Check distance from existing harbors
            let tooClose = false;
            for (const harbor of harbors) {
                const dist = Math.hypot(pos.worldX - harbor.x, pos.worldY - harbor.y);
                if (dist < minDistance) {
                    tooClose = true;
                    break;
                }
            }
            
            if (!tooClose) {
                const islandStub = {
                    id: `island_${harbors.length}`,
                    x: pos.worldX,
                    y: pos.worldY,
                    radius: 50
                };
                const harbor = new Harbor(`harbor_${harbors.length}`, islandStub);
                harbors.push(harbor);
            }
        }
        
        console.log(`Generated ${harbors.length} harbors on land edges (tile-based, temporary)`);
        return harbors;
    }
